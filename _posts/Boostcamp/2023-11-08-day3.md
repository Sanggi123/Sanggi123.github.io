```python
import numpy as np
```

```python
a = [1, 2, 3, 4, 5]
b = [5, 4, 3, 2, 1]

print ('python list : ', a[0] is b[-1])

a = np.array(a)
b = np.array(b)

print ('ndarray : ', a[0] is b[-1])
```

    python list :  True
    ndarray :  False

```python
c = np.array([[1,2,3,],[4,5,6]])
print('shape : ', c.shape)
print('ndim : ', c.ndim) # 차원
print('size : ', c.size) # 개수
```

    shape :  (2, 3)
    ndim :  2
    size :  6
